Javascript


netive.rangeSlider.init({
	id : 'range1'
});

//single type 필수
netive.rangeSlider.rangeFrom({ 
	id: 'range1', 
	value: 90,
	type: 'single'
});

//dual type 생략
netive.rangeSlider.rangeFrom({ 
	id: 'range1', 
	value: 90
});
netive.rangeSlider.rangeTo({ 
	id: 'range1', 
	value: 90 
});
