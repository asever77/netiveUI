Javascript


netive.rangeSlider.init({
	id: 'range1',
	step: 500,
	min:1000,
	max:5000,
	text: ['매우높음', '매우높음2', '높음', '높음2', '보통', '보통2','낮음', '낮음2', '매우낮음'],
	tickmark: ['1000', '2000', '3000', '4000', '5000'],
});

//single type 필수
netive.rangeSlider.rangeFrom({ 
	id: 'range1', 
	value: 90,
	type: 'single'
});

//dual type 생략
netive.rangeSlider.rangeFrom({ 
	id: 'range1', 
	value: 90
});
netive.rangeSlider.rangeTo({ 
	id: 'range1', 
	value: 90 
});
