Javascript


//생성
netive.datepicker.week = ['일', '월', '화', '수', '목', '금', '토', '년', '월', '일'];
netive.datepicker.isFooter = false; true - 하단 확인버튼 생성, false 하단 버튼 없음. 날짜 클릭 시 바로 선택

netive.datepicker.init({
	id: 'input id name',
	date: 'YYYY-MM-DD',
	min: 'YYYY-MM-DD',
	max: 'YYYY-MM-DD',
	title: 'title name',
	period: 'start' or 'end',
	callback: function(){
		console.log('callback init');
	}
});

//확인
netive.datepicker.confirm({
	id: 'input id name',
	callback: function(){
		console.log('callback confirm');
	}
});

//삭제
netive.datepicker.destoroy({
	id: 'input id name',
	callback: function(){
		console.log('callback destoroy');
	}
});

//callback
netive.callback.{datepicker id} = (v) => {
	console.log('select date value:', v);
}
