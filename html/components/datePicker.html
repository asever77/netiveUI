Javascript


//생성
$plugins.datepicker.init({
	id: 'input id name',
	date: 'YYYY-MM-DD',
	min: 'YYYY-MM-DD',
	max: 'YYYY-MM-DD',
	title: 'title name',
	period: 'start' or 'end',
	callback: function(){
		console.log('callback init');
	}
});

//확인
$plugins.datepicker.confirm({
	id: 'input id name',
	callback: function(){
		console.log('callback confirm');
	}
});

//삭제
$plugins.datepicker.destoroy({
	id: 'input id name',
	callback: function(){
		console.log('callback destoroy');
	}
});
