Source Code

HTML


//스크립트 적용 전
<div class="ui-scrollbar">
	..........
</div>

//스크립트 적용 후
<div class="ui-scrollbar" scroll-id="uiScrollBar_0">
	<div class="ui-scrollbar-barwrap type-y">
		<button type="button" class="ui-scrollbar-bar" aria-hidden="true" tabindex="-1" data-scrollxy="y" style="...">
			<span class="hide">scroll</span>
		</button>
	</div>
	<div class="ui-scrollbar-barwrap type-x">
		<button type="button" class="ui-scrollbar-bar" aria-hidden="true" tabindex="-1" data-scrollxy="x" style="...">
			<span class="hide">scroll</span>
		</button>
	</div>
	<div class="ui-scrollbar-item ready">
		<div class="ui-scrollbar-wrap">
			..........
		</div>
	</div>
</div>

CSS

Javascript


//전체 적용 
$plugins.scrollbar.init({
	callback: function(){
		console.log('end');
	}
});

//개별 적용
$plugins.scrollbar.init({
	id: '{name}',
	callback: function(){
		console.log('end');
	}
});
옵션 설명
Options Type Default Explanation
id: [string] scroll-id의 값
callback: [function] function y축 스크롤이 젤하단에 도착했을 경우 콜백 실행
