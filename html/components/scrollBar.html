<div class="ui-scrollbar" id="uiScrollBar_0">
	<div class="ui-scrollbar-barwrap">
		<button type="button" class="ui-scrollbar-bar" aria-hidden="true" tabindex="-1" style="height: 26%;">
			<span class="hide">스크롤버튼</span>

</div>

$plugins.uiScrollBar


//전체 적용 .ui-scrollbar

//개별 적용

	id: 'id name',
	top: 0
});
