Javascript


$plugins.rangeSlider.init({
	id : 'range1'
});

//single type 필수
$plugins.rangeSlider.rangeFrom({ 
	id: 'range1', 
	value: 90,
	type: 'single'
});

//dual type 생략
$plugins.rangeSlider.rangeFrom({ 
	id: 'range1', 
	value: 90
});
$plugins.rangeSlider.rangeTo({ 
	id: 'range1', 
	value: 90 
});
