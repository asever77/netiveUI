<a class="exe-winpop" href="..." target="_blank" aria-haspopup="true" title="새 창 열림">팝업</a>
// 필수요소 : 
class ="ui-winpop", 
href="...", 
// 선택요소 : 
data-name
data-width
data-height
data-align
data-top
data-left
data-toolbar
data-location
data-menubar
data-status
data-resizable
data-scrollbars

$(documnet).off('click.winpop').on('click.winpop', '.exe-winpop', function(e){
	e.preventDefault()
	$plugins.uiPopup({
		link: this.href,
		name: $(this).data('name'),
		width: $(this).data('width'),
		height: $(this).data('height'),
		align: $(this).data('align'),
		top: $(this).data('top'),
		left: $(this).data('left'),
		toolbar: $(this).data('toolbar'),
		location: $(this).data('location'),
		menubar: $(this).data('menubar'),
		status: $(this).data('status'),
		resizable: $(this).data('resizable'),
		scrollbars: $(this).data('scrollbars')
	});
});
