Javascript


netive.scroll.parallax({
	callback:function(v) {
		console.log(v);
	}
});

netive.callback.실행함수명 = function(opt){
	const el = opt.el;
	const n = opt.n;
	
	//스타일 적용
	el.style.opacity = n;
	el.style.transform = 'translate('+ ((100 - n * 100).toFixed(2)) +'% ,0)';
}
