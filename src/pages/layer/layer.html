Layer Contents

z축의 위치로 구분되는 레이어 콘텐츠입니다.
사용자에게 선택 행동을 요구하는 modal, alert, bottomsheet, dropdown 콘텐츠와
일시적인 알림, 간단한 정보제공을 하는 toast, tooltip 콘텐츠로 분류합니다.

Modal & Bottomsheet & Dropdown Layer


<section class="mdl-layer" data-id="{{ name }}" data-type="{{ modal | bottom | drowdown }}">
    <div class="mdl-layer-wrap">
        <div class="mdl-layer-header">
            <h1>제목...</h1>
            <button type="button" class="mdl-layer-close" aria-label="닫기"></button>
        </div>
        <div class="mdl-layer-body" style="height: 100vh;">
            내용...
        </div>
        <div class="mdl-layer-footer">
            <button type="button" class="mdl-button" data-toggle-object="c">
                <span>확인</span>
            </button>
        </div>
    </div>
    <div class="mdl-layer-dim"></div>
</section>

<script type="module">
    import Layer from '../assets/js/components/layer.js';
    
    // Modal Layer
    UI.exe.modal_1 = new Layer({
        id :'{{ name }}',
        type: 'modal',
        src: '{{ ./route/filename }}'
    });
    UI.exe.modal_1.show();
    UI.exe.modal_1.hide();

    // Bottom Sheet Layer
    UI.exe.bottom_1 = new Layer({
        id :'{{ name }}',
        type: 'bottom',
        src: '{{ ./route/filename }}'
    });
    UI.exe.bottom_1.show();
    UI.exe.bottom_1.hide();

    // dropdown Layer
    UI.exe.dropdown1 = new Layer({
        id :'{{ name }}',
        type: 'dropdown',
        ps : 'BL',
        src: '{{ ./route/filename }}'
    });
    UI.exe.dropdown1.show();
    UI.exe.dropdown1.hide();
</script>

System Layer


//자동생성
<section class="mdl-layer" data-id="alert1" data-type="alert" data-state="show" data-layer-current="true" data-layer-n="1" style="z-index: 101;">
    <div class="mdl-layer-wrap">
        <div class="mdl-layer-body">
            <h1 class="mdl-layer-tit">{{ title }}</h1>
            <div>{{ content }}</div>
            <div class="mdl-btn-wrap">
                <button type="button" class="mdl-btn" data-state="cancel" data-style="primary-gray">
                    <span>{{ button[1].text }}</span>
                </button>
                <button type="button" class="mdl-btn" data-state="ok" data-style="primary"> 
                    <span>{{ button[0].text }}</span>
                </button> 
            </div>
        </div>
    </div>
    <div class="mdl-layer-dim"></div>
</section>

<script type="module">
    import Layer from '../assets/js/components/layer.js';
    
    // System Layer
    UI.exe.alert_1 = new Layer({
        id :'alert1',
        type: 'system',
        title: 'alert title',
        content: '안녕하세요 저장이 완료되었습니다.',
        button: [
            {text: '확인', callback: alertCallbackOk},
            {text: '취소', callback: alertCallbackCancel}
        ]
    });
    UI.exe.alert_1.show();
    UI.exe.alert_1.hide();
</script>
